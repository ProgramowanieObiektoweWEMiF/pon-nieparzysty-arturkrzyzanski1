    private void loginActionPerformed(java.awt.event.ActionEvent evt) {                                      
        String loginn = this.jTextField1.getText();
        String hasloo = String.valueOf(this.haslo.getPassword());
       //String hasloo =this.haslo.getText();
        if(b.checkUzytkownik(loginn, hasloo))
        {
            Uzytkownik u = b.selectUzytkownik(loginn, hasloo);
          if(u.getRola().equals("uzytkownik"))
            {
            UzytkownikForm uf = new UzytkownikForm(u);
            uf.setVisible(true);
            uf.pack();
            uf.setLocationRelativeTo(null);
            uf.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            b.closeConnection();
            this.dispose();  
          }else if(u.getRola().equals("admin"))
          {
            AdminForm af = new AdminForm(u);
            af.setVisible(true);
            af.pack();
            af.setLocationRelativeTo(null);
            af.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            b.closeConnection();
            this.dispose();
            
          }
        }
        else
        {
           System.out.println("niema");
           this.jTextField1.setText("");
           this.haslo.setText("");
           JOptionPane.showMessageDialog(this, "Niepoprawny logi lub has³o.");
        }
        
    }
