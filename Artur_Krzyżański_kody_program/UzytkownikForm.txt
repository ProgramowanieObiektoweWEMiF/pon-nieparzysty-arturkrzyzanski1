    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        
        SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd");
        GregorianCalendar data =(GregorianCalendar) this.data_start.getCalendar();
        GregorianCalendar data2 =(GregorianCalendar) this.data_stop.getCalendar();
        SimpleDateFormat timeFormat = new SimpleDateFormat("HH:mm:ss");
        Date czas =(Date) this.godzina_start.getValue();
        Date czas2 =(Date) this.godzina_stop.getValue();
        String data_start = dateFormat.format(data.getTime());
        String data_stop = dateFormat.format(data2.getTime());
        String czas_start=timeFormat.format(czas.getTime());
        String czas_stop=timeFormat.format(czas2.getTime());
        String start=data_start+" "+czas_start;
        String stop=data_stop+" "+czas_stop;
        DefaultTableModel model = (DefaultTableModel) jTable1.getModel();
        int id_s;
        if(data!=null && data2!=null)
        {
            if (start.compareTo(stop)<0)
            {                
                System.out.println(model.getDataVector().elementAt(jTable1.getSelectedRow()));
                id_s = (int) model.getValueAt(jTable1.getSelectedRow(), 0);
                System.out.println(id_s);
                b.insertZamowienie(id_s, u.getID_uzytkownik(), start, stop, "zarezerwowano");
                try {
                    b.updateSamochod(String.valueOf(id_s));
                } catch (SQLException ex) {
                    Logger.getLogger(UzytkownikForm.class.getName()).log(Level.SEVERE, null, ex);
                }
                //System.out.println(id_s+" "+u.getID_uzytkownik()+" "+start+" "+stop+" "+"zarezerwowano");
                uzupelnij();
            }
            else
            {
                jTextPane1.setText("Niepoprawna data");
                JOptionPane.showMessageDialog(this, "Data wypo¿yczenia musi byæ mniejsza ni¿ data zwrotu");
            }
       
        }
        else
        {
        System.out.println("null");
        }  
        uzupelnij2();
    }

    void uzupelnij()
    {
        ArrayList<Zamowienie> zamowienia = new ArrayList<Zamowienie>();
        zamowienia = b.selectZamowienia();
       // jTable1
        ArrayList<Samochod> samochody = new ArrayList<Samochod>();
        
        samochody = b.selectSamochody();    
        DefaultTableModel model = (DefaultTableModel) jTable1.getModel();
        model.setRowCount(0);
        Object rowData[] = new Object[5];
        for(int i =0; i<samochody.size();i++)
        {
            if(samochody.get(i).getStatus().equals("wolny"))
            {
            rowData[0] = samochody.get(i).getID_samochod();
            rowData[1] = samochody.get(i).getMarka();
            rowData[2] = samochody.get(i).getRok();
            rowData[3] = samochody.get(i).getCena();
            rowData[4] = samochody.get(i).getStatus();
            model.addRow(rowData);
            }
        }
    }
