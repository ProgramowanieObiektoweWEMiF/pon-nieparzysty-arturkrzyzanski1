    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        ArrayList<Samochod> samochody = new ArrayList<Samochod>();
        samochody=b.selectSamochody();
       
 jTextPane1.setText("");
        System.out.println(samochody.size());

        for(int i=0; i<samochody.size();i++)
        {
            
            if(jCheckBox1.isSelected())
            {
                if(samochody.get(i).getStatus().equals("wolny"))
                {
                jTextPane1.setText(jTextPane1.getText()+samochody.get(i).toString()+"----------------------------\n\n");
                    
                }
            }
            if(jCheckBox2.isSelected())
            {
                if(samochody.get(i).getStatus().equals("zarezerwowany"))
                {
                jTextPane1.setText(jTextPane1.getText()+samochody.get(i).toString()+"----------------------------\n\n");
                }
            }
            if(jCheckBox3.isSelected())
            {
                if(samochody.get(i).getStatus().equals("wypozyczony"))
                {
                jTextPane1.setText(jTextPane1.getText()+samochody.get(i).toString()+"----------------------------\n\n");
                }
            }
        }
    }
