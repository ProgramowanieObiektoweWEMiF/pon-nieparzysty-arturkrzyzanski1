            ArrayList<Uzytkownik> uzytkownicy = new ArrayList<Uzytkownik>();
            uzytkownicy = b.selectuzytkownicy();
            int liczba = uzytkownicy.size();
            boolean takiSamL=false;

                    takiSamL=true;
                    break;
                }
            }
            if(jTextPane1.getText().isEmpty() || jTextPane2.getText().isEmpty() || jTextPane3.getText().isEmpty() || jTextPane4.getText().isEmpty() || takiSamL)
            {
                System.out.println("puste pole");
                System.out.println("Taki sam login " +takiSamL);
                JOptionPane.showMessageDialog(this, "Login jest ju¿ zajêty, podaj inny.");
                ;
            }
            else
            {
                if(jTextPane3.getText().equals("admin") || jTextPane3.getText().equals("uzytkownik"))
                {

                    try {
                        b.updateUzytkownik(id_u_edytuj, jTextPane1.getText(), jTextPane2.getText(), jTextPane3.getText());
                    } catch (SQLException ex) {
                        Logger.getLogger(AdminForm.class.getName()).log(Level.SEVERE, null, ex);
                    }
                    uzupelnijuzytkownikow();
                    jTextPane1.setText("");
                    jTextPane2.setText("");

                }
            }

        }

    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        int row = jTable2.getSelectedRow();
        if(row == -1)
        {
            JOptionPane.showMessageDialog(this, "Nie wybrano zamówienia, proszê wybierz zamówienie.");
        }
        else
        {
            DefaultTableModel model = (DefaultTableModel) jTable2.getModel();
            Vector v = (Vector) model.getDataVector().elementAt(jTable2.getSelectedRow());

            String z = String.valueOf(v.get(0));
            String s = String.valueOf(v.get(1));
            String status = String.valueOf(v.get(3));
            String c = (String) jComboBox1.getSelectedItem();
            if(jTextField4.getText().isEmpty())
            {
                JOptionPane.showMessageDialog(this, "Nie wybrano zamówienia, proszê wybierz zamówienie.");
            }
            else if(jComboBox1.getSelectedItem().equals(status))
            {
                JOptionPane.showMessageDialog(this, "Wybrany status jest taki sam jak obecny.");
            }
            else
            {
                try {
                    b.updateZamowieie(z, c);
                    if(c.equals(combo[0]))
                    {
                        b.updateSamochod(s, "zarezerwowany");
                    }
                    else if(c.equals(combo[1]))
                    {
                        b.updateSamochod(s, "wypozyczony");
                    }
                    else
                    {
                        b.updateSamochod(s, "wolny");
                    }
                    uzupelnij2();
                    uzupelnijSamochody();
                    uzupelnijuzytkownikow();
                } catch (SQLException ex) {
                    Logger.getLogger(AdminForm.class.getName()).log(Level.SEVERE, null, ex);
                }
            }
        }
    }
